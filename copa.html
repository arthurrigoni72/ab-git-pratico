let x, y, vx, vy; let interval; let running = false; let scoreA = 0; let scoreB = 0; let time = 0; let finished = false; const ball = document.getElementById("ball"); function resetBall() { x = 300; y = 150; vx = (Math.random() * 3 + 2) * (Math.random() < 0.5 ? -1 : 1); vy = (Math.random() * 3 + 2) * (Math.random() < 0.5 ? -1 : 1); } /* 🎯 controle de chance de gol (IMPORTANTE) */ function chanceGol() { return Math.random() < 0.012; // baixa chance → média ~1–2 gols por time } function updateBall() { const w = 600; const h = 300; x += vx; y += vy; // quique em cima/baixo if (y <= 0 || y >= h - 18) vy *= -1; /* ⚽ tentativa de gol (controlada) */ if (chanceGol() && x < 30) { scoreB++; log("⚽ GOL DO TIME B!"); resetBall(); } if (chanceGol() && x > w - 30) { scoreA++; log("⚽ GOL DO TIME A!"); resetBall(); } ball.style.left = x + "px"; ball.style.top = y + "px"; } function log(msg) { document.getElementById("log").innerHTML = msg; } function updateScore() { document.getElementById("score").innerText = `${scoreA} x ${scoreB}`; } function startMatch() { if (running) return; scoreA = 0; scoreB = 0; time = 0; finished = false; log("🔥 Partida iniciada!"); updateScore(); resetBall(); running = true; interval = setInterval(() => { updateBall(); updateScore(); time++; /* ⏱️ 30 segundos reais */ if (time >= 3000) { endMatch(); } }, 10); } function endMatch() { clearInterval(interval); running = false; if (scoreA === scoreB) { log("🥶 Empate! Indo para pênaltis..."); setTimeout(penaltis, 1200); } else { const winner = scoreA > scoreB ? document.getElementById("a").value : document.getElementById("b").value; log(`🏆 Fim de jogo! ${winner} venceu`); } finished = true; } function penaltis() { let pa = 0, pb = 0; for (let i = 0; i < 5; i++) { if (Math.random() < 0.7) pa++; if (Math.random() < 0.7) pb++; } if (pa === pb) { if (Math.random() > 0.5) pa++; else pb++; } const winner = pa > pb ? document.getElementById("a").value : document.getElementById("b").value; log(`🥅 Pênaltis: ${pa} x ${pb} — 🏆 ${winner}`); } document.addEventListener("keydown", (e) => { if (e.key === "Enter" && finished) { log("Pronto para próxima partida"); } });
